SELECT DISTINCT l.idPedido, l.idProducto as prod, nombre, SUM(cantidad) AS cant, t.descripcion, pr.precio as precioIn, (pr.precio*SUM(cantidad)) as precioFin FROM lineapedido l, pedido p, productos pr, tipo t WHERE pr.idProducto=l.idProducto and cliente='$idUs' and pr.tipo=t.idTipo GROUP BY l.idProducto ORDER BY t.descripcion;
